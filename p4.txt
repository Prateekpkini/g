import gensim.downloader as api
from transformers import AutoTokenizer, AutoModelForSeq2SeqLM
print("Loading Word Embedding Model...")
word_model = api.load("glove-wiki-gigaword-100")
print("Model Loaded ✅")
original_prompt = "Explain the role of Artificial Intelligence in healthcare."
print("\nOriginal Prompt:")
print(original_prompt)
keywords = ["artificial_intelligence", "healthcare"]
similar_words = []
for word in keywords:
    if word in word_model:
        similar = word_model.most_similar(word, topn=3)
        for w in similar:
            similar_words.append(w[0])
print("\nSimilar Words Retrieved:")
print(similar_words)
enriched_prompt = original_prompt + \
"\nInclude topics like: " + ", ".join(similar_words) + "."
print("\nEnriched Prompt:")
print(enriched_prompt)
print("\nLoading Generative AI Model...")
tokenizer = AutoTokenizer.from_pretrained("google/flan-t5-small")
model = AutoModelForSeq2SeqLM.from_pretrained("google/flan-t5-small")
print("GenAI Model Loaded ✅")
def generate_response(prompt):
    inputs = tokenizer(prompt, return_tensors="pt")
    outputs = model.generate(**inputs, max_length=200)
    return tokenizer.decode(outputs[0], skip_special_tokens=True)
response_original = generate_response(original_prompt)
response_enriched = generate_response(enriched_prompt)
print("\n===== ORIGINAL PROMPT RESPONSE =====\n")
print(response_original)
print("\n===== ENRICHED PROMPT RESPONSE =====\n")
print(response_enriched)
print("\n===== COMPARISON =====")
print(f"\nResponse Length:")
print(f"Original : {len(response_original)}")
print(f"Enriched : {len(response_enriched)}")
